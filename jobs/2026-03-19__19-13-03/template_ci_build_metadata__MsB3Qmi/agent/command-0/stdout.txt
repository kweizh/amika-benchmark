import json, subprocess, pathlib
repo = '/home/user/repo'
out = pathlib.Path('/home/user/ci_metadata.json')

def jj(template, rev='@'):
    return subprocess.check_output(['jj','log','--no-graph','-r',rev,'-T',template], cwd=repo, text=True)

rev = '@'
if jj('empty', rev).strip() == 'true':
    rev = '@-'

commit_id = jj('commit_id', rev).strip()
change_id = jj('change_id', rev).strip()
author = jj('author.name()', rev).strip()
email = jj('author.email()', rev).strip()
timestamp = jj('committer.timestamp().format("%Y-%m-%dT%H:%M:%S")', rev).strip()
description = jj('description', rev).rstrip('\n')
parents_raw = jj('parents.map(|p| p.change_id().short(12)).join(",")', rev).strip()
parents = [p for p in parents_raw.split(',') if p]

data = {
    'commit_id': commit_id,
    'change_id': change_id,
    'author': author,
    'email': email,
    'timestamp': timestamp,
    'description': description,
    'parents': parents,
}
out.write_text(json.dumps(data, indent=2) + '\n')
print(rev)
print(out)
