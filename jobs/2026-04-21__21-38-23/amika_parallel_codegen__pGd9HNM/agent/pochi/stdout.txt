#!/bin/bash

# Create sandboxes
amika sandbox create --name feature-a --git
amika sandbox create --name feature-b --git

# Send prompts in parallel
amika sandbox agent-send feature-a "Generate frontend components" &
PID_A=$!

amika sandbox agent-send feature-b "Generate backend API" &
PID_B=$!

# Wait for both background processes to complete
