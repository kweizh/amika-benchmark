import json, subprocess
repo='/home/user/repo'
def jj(template):
    return subprocess.check_output(['jj','log','--no-graph','-r','@','-T',template], cwd=repo, text=True).rstrip('\r\n')
parents_raw = jj('parents.map(|p| p.change_id().short(12)).join(",")')
parents = [p for p in parents_raw.split(',') if p]
data = {
  'commit_id': jj('commit_id'),
  'change_id': jj('change_id'),
  'author': jj('author.name()'),
  'email': jj('author.email()'),
  'timestamp': jj('committer.timestamp().format("%Y-%m-%dT%H:%M:%S")'),
  'description': jj('description'),
  'parents': parents,
}
with open('/home/user/ci_metadata.json','w', encoding='utf-8') as f:
    json.dump(data, f, indent=2)
    f.write('\n')
